/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int maxPathSum(TreeNode root) {
        int[] mx = new int[1];
        mx[0]=Integer.MIN_VALUE;
        mxt(root, mx);
        return mx[0];
    }
    public int mxt(TreeNode root, int[] mx){
        if(root==null){
            return 0;
        }
        int lh=Math.max(0, mxt(root.left, mx));
        int rh=Math.max(0, mxt(root.right, mx));
        mx[0]=Math.max(mx[0], lh+rh+root.val);
        return Math.max(lh,rh)+root.val;
    }
}
