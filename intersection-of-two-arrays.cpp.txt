class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        map<int,int> mp;
        for(int i:nums1){
            mp[i]++;
        }
        set<int> ans;
        for(int j:nums2){
            if(mp[j]>0){
                ans.insert(j);
            }
        }
        vector<int> ans2;
        for(auto it:ans){
            ans2.push_back(it);
        }
        return ans2;
    }
};
