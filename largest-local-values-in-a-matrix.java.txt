class Solution {
    public int[][] largestLocal(int[][] grid) {
        int si=grid.length;
        int[][] ans = new int[si-2][si-2];
        for(int k=0;k<si-2;k++){
            for(int p=0;p<si-2;p++){
                for(int i=k;i<k+3;i++){
                    for(int j=p;j<p+3;j++){
                        ans[k][p]=Math.max(ans[k][p],grid[i][j]);
                    }
                }
            }
        }
        return ans;
    }
}
