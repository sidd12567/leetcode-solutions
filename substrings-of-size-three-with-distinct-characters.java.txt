class Solution {
    public int countGoodSubstrings(String s) {
        Set<Character> st = new HashSet<>();
        int ans=0;
        for(int i=0;i<s.length()-2;i++){
            for(int j=0;j<3;j++){
                st.add(s.charAt(i+j));
            }
            if(st.size()==3){
                ans++;
            }
            st.clear();
        }
        return ans;
    }
}
