class Solution {
public:
    vector<vector<int>> matrixReshape(vector<vector<int>>& mat, int r, int c) {
        vector<vector<int>> ans(r,vector<int> (c,0));
        int ro=mat.size();
        int co=mat[0].size();
        int tc=r*c;
        int tr=ro*co;
        if(tc==tr){
            for(int i=0;i<tr;i++){
                ans[i/c][i%c]=mat[i/co][i%co];
            }
            return ans;
        }
        return mat;
    }
};
