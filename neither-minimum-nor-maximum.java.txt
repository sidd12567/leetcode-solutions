class Solution {
    public int findNonMinOrMax(int[] nums) {
        Arrays.sort(nums);
        int mn=nums[0];
        int si=nums.length;
        int mx=nums[si-1];
        for(int i=0;i<si;i++){
            if(mn!=nums[i] && mx!=nums[i]){
                return nums[i];
            }
        }
        return -1;
    }
}
