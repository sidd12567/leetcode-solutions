class Solution {
public:
    vector<string> sortPeople(vector<string>& names, vector<int>& heights) {
        priority_queue<pair<int,string>> pq;
        int si=names.size();
        vector<string> ans;
        for(int i=0;i<si;i++){
            pq.push({heights[i],names[i]});
        }
        while(!pq.empty()){
            pair<int,string> po=pq.top();
            ans.push_back(po.second);
            pq.pop();
        }
        return ans;
    }
};
