class Solution {
public:
    bool judgeCircle(string moves) {
        int uc=0,dc=0,lc=0,rc=0;
        for(char c:moves){
            if(c=='U'){
                uc++;
            }
            if(c=='D'){
                dc--;
            }
            if(c=='L'){
                lc++;
            }
            if(c=='R'){
                rc--;
            }
        }
        if((uc+dc==0)&& (rc+lc==0)){
            return true;
        }
        return false;
    }
};
