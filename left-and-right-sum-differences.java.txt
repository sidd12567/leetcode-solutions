class Solution {
    public int[] leftRightDifference(int[] nums) {
        int sm=Arrays.stream(nums).sum();
        int l=nums.length;
        int ls=0;
        int[] ans = new int[l];
        for(int i=0;i<l;i++){
            int tmp=sm-nums[i]-ls;
            int smm=Math.abs(tmp-ls);
            ans[i]=smm;
            ls+=nums[i];
        }
        return ans;
    }
}
