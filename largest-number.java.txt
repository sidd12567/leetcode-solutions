class Solution {
    public String largestNumber(int[] nums) {
        int l=nums.length;
        String[] str = new String[l];
        for(int i=0;i<l;i++){
            str[i]=String.valueOf(nums[i]);
        }
        Arrays.sort(str, (a,b)->(b+a).compareTo(a+b));
        if(str[0].equals("0")){
            return "0";
        }
        StringBuilder sb = new StringBuilder();
        for(String s:str){
            sb.append(s);
        }
        return sb.toString();
    }
}
