class Solution {
    public int searchInsert(int[] nums, int target) {
        int nf=0, ans=0;
        for(int i=0;i<nums.length;i++){
            if(nums[i]==target){
                ans=i;
                return ans;
            }
            else if(nums[i]<target){
               nf=i+1;
            }
            
        }
        return nf;
    }
}
