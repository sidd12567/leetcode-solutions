class Solution {
    public long dividePlayers(int[] skill) {
        Arrays.sort(skill);
        int l=skill.length;
        long ans=0;
        int pr=skill[0]+skill[l-1];
        for(int i=0;i<l/2;i++){
            if(skill[i]+skill[l-i-1]!=pr){
                return -1;
            }
            ans+=(skill[i]*skill[l-i-1]);
    }
    return ans;
}
}
