class Solution {
public:
    vector<int> decompressRLElist(vector<int>& nums) {
        vector<int> ans;
        int s=nums.size();
        for(int i=0;i<nums.size();i++){
            int v,f;
            v=2*i;
            f=2*i+1;
            if(f<s && v<s){
                v=nums[v];
                f=nums[f];
                while(v--){
                    ans.push_back(f);
                }
            }
        }
        return ans;
    }
};
