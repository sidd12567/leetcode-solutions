class Solution {
    public int divisorSubstrings(int num, int k) {
        int ans=0;
        String s=Integer.toString(num);
        int si=s.length();
        for(int i=0;i<si-k+1;i++){
            String tmp="";
            for(int j=0;j<k;j++){
                tmp+=s.charAt(i+j);
            }
            int ti=Integer.parseInt(tmp);
            if(ti!=0){
                if(num%ti==0){
                    ans++;
                }
            }
        }
        return ans;
    }
}
