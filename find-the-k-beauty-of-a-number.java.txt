class Solution {
    public int divisorSubstrings(int num, int k) {
        int ans=0;
        String s=Integer.toString(num);
        String t="";
        int l=s.length();
        for(int i=0;i<l-k+1;i++){
            for(int j=i;j<i+k;j++){
                t+=s.charAt(j);
            }
            int n=Integer.parseInt(t);
            if(n!=0){
                if(num%n==0){
                    ans++;
                }
            }
            t="";
        }
        return ans;
    }
}
