class Solution {
    public int minOperations(int[] nums) {
        HashMap<Integer, Integer> mp = new HashMap<>();
        int ans=0;
        for(int i:nums){
            mp.put(i, mp.getOrDefault(i, 0)+1);
        }   
        for(int i:mp.keySet()){
            int k=mp.get(i);
            if(k==1){
                return -1;
            }
            ans+=Math.ceil((double)k/3);
        }
        return ans;
    }
}
