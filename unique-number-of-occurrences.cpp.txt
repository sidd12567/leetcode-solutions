class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        map<int,int> mp;
        for(int i:arr){
            mp[i]++;
        }
        set<int> ans;
        for(auto it:mp){
            ans.insert(it.second);
        }
        if(ans.size()!=mp.size()){
            return false;
        }
        return true;
    }
};
