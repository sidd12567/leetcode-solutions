class Solution {
    public int maximumStrongPairXor(int[] nums) {
        int ans=0;
        int l=nums.length;
        Arrays.sort(nums);
        for(int i=0;i<l;i++){
            for(int j=i+1;j<l;j++){
                if(Math.abs(nums[i]-nums[j])<=Math.min(nums[i],nums[j])){
                    ans=Math.max(ans,nums[i]^nums[j]);
                }
                
            }
        }
        return ans;
    }
}
