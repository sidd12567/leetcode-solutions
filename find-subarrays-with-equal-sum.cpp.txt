class Solution {
public:
    bool findSubarrays(vector<int>& nums) {
        map<int,int> mp;
     for(int i=0;i<nums.size()-1;i++){
         
             int s=nums[i]+nums[i+1];
             mp[s]++;
             if(mp[s]>1){
                 return true;
             }
         
     }
        return false;
    }
};
