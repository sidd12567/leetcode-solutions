class Solution {
    public int differenceOfSum(int[] nums) {
        int ans=0;
        for(int i:nums){
            ans+=i;
        }
        int ani=0;
        for(int j=0;j<nums.length;j++){
            String t = Integer.toString(nums[j]);
            for(int k=0;k<t.length();k++){
                ani+=(t.charAt(k)-'0');
            }
        }
        return Math.abs(ans-ani);
    }
}
