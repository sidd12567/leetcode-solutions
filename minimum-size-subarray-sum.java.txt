class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int ans=Integer.MAX_VALUE;
        int i=0,sm=0;
        for(int j=0;j<nums.length;j++){

            sm+=nums[j];
            while(sm>=target){
                ans=Math.min(ans,j-i+1);
                sm-=nums[i];
                i++;
            }
        }
        return ans==Integer.MAX_VALUE?0:ans;
    }
}
