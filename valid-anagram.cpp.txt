class Solution {
public:
    bool isAnagram(string s, string t) {
        map<char,int> mp,mp1;
        for(char c:s){
            mp[c]++;
        }
        for(char c:t){
            mp1[c]++;
        }
        for(auto it:mp1){
            if(mp[it.first]!=it.second){
                return false;
            }
        }
        for(auto it:mp){
            if(mp1[it.first]!=it.second){
                return false;
            }
        }
        return true;
    }
};
