class Solution {
public:
    int maxPower(string s) {
        int ans=1,mx=1;
        for(int i=1;i<s.size();i++){
            if(s[i]==s[i-1]){
                ans++;
                mx=max(ans,mx);
            }
            else{
                mx=max(mx,ans);
                ans=1;
            }
        }
        return mx;
    }
};
