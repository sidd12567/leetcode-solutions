class Solution {
    public int alternateDigitSum(int n) {
        int ans=0;
        String a=String.valueOf(n);
        for(int i=0;i<a.length();i++){
            if(i%2==0){
                ans+=(a.charAt(i)-'0');
            }
            else{
                ans-=(a.charAt(i)-'0');
            }
        }
        return ans;
    }
}
