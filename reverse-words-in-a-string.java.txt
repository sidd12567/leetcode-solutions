class Solution {
    public String reverseWords(String s) {
        String[] ans = s.split("\\s+");
        int l=ans.length;
        String ans2="";
        for(int i=0;i<l;i++){
            
            ans2+=ans[l-i-1];
            ans2+=" ";
        }
        ans2=ans2.trim();
        return ans2;
    }
}
