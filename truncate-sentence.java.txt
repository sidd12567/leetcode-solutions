class Solution {
    public String truncateSentence(String s, int k) {
        String[] ans=s.split(" ");
        String ani="";
        for(int i=0;i<k;i++){
            ani+=ans[i];
            ani+=" ";
        }
        int l=ani.length();
        ani=ani.substring(0,l-1);
        return ani;
    }
}
