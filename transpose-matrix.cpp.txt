class Solution {
public:
    vector<vector<int>> transpose(vector<vector<int>>& matrix) {
        int si=matrix.size();
        int ss=matrix[0].size();
        vector<vector<int>> ans(ss,vector<int> (si));
        for(int i=0;i<si;i++){
            for(int j=0;j<ss;j++){
                ans[j][i]=matrix[i][j];
            }
        }
        return ans;
    }
};
