class Solution {
    public int minimumMoves(String s) {
        String t="";
        int ans=0;
        
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)!='O'){
                ans++;
                i+=2;
                t+="OOO";
            }
            else{
                t+=s.charAt(i);
            }
        }
        return ans;
    }
}
