class Solution {
public:
    vector<int> getConcatenation(vector<int>& nums) {
        vector<int> ans(2*nums.size());
        for(int i=0;i<nums.size();i++){
            ans[i]=nums[i];
        }
        int s=nums.size();
        for(int i=s;i<2*s;i++){
            ans[i]=ans[i-s];
        }
        return ans;
    }
};
