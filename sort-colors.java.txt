class Solution {
    public void sortColors(int[] nums) {
        int ar[]= new int[3];
        for(int i=0;i<nums.length;i++){
            ar[nums[i]]++;
        }
        int tmp=0,t=0;
        for(int i=0;i<3;i++){
            for(int j=0;j<ar[i];j++){
                nums[tmp]=t;
                tmp++;
            }
            t++;
        }
    }
}
