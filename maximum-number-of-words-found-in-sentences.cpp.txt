class Solution {
public:
    int mostWordsFound(vector<string>& sentences) {
        
        int cnt=0;
        int mx=0;
        for(auto it:sentences){
            for(auto ij:it){
                if(ij==' '){
                    cnt++;
                }
            }
            mx=max(mx,cnt+1);
            cnt=0;
        }
        
        return mx;
    }
};
