class Solution {
    public int countPairs(List<Integer> nums, int target) {
        int ans=0;
        int si=nums.size();
        for(int i=0;i<si;i++){
            for(int j=i+1;j<si;j++){
                if(nums.get(i)+nums.get(j)<target){
                    ans++;
                }
            }
        }
        return ans;
    }
}
