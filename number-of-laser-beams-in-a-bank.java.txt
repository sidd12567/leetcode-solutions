class Solution {
    public int numberOfBeams(String[] bank) {
        int ans=0;
        int cnt=0;
        int pr=0;
         for(int i=0;i<bank.length;i++){
            String t=bank[i];
            for(int j=0;j<t.length();j++){
                if(t.charAt(j)=='1'){
                    cnt++;
                }
            }
            if(cnt==0){
                continue;
            }
            ans=ans+cnt*pr;
            pr=cnt;
            cnt=0;
        }
        return ans;

    }
}
