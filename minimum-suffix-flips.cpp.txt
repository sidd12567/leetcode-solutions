class Solution {
public:
    
    
    int minFlips(string target) {
        int si=target.size();
        int ans=0;
        char st='0';
        for(int i=0;i<si;i++){
            if(st!=target[i]){
                ans++;
                st=(st=='0')?'1':'0';
            }
            
        }
    return ans;
    }
};
