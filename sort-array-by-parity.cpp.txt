class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& nums) {
        int si=nums.size();
        vector<int> ans(si);
        int cnt=0;
        for(int i:nums){
            if(i%2==0){
                ans[cnt++]=i;
            }
        }
        for(int i:nums){
            if(i%2!=0){
                ans[cnt++]=i;
            }
        }
        return ans;
        
    }
};
