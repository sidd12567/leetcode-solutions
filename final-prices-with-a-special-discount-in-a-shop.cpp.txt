class Solution {
public:
    vector<int> finalPrices(vector<int>& prices) {
        int si=prices.size();
        stack<int> st;
        for(int i=si-1;i>=0;i--){
            while(!st.empty() && st.top()>prices[i]){
                st.pop();
            }
            int u=prices[i];
            if(!st.empty()){
                prices[i]-=st.top();
               
            }
            st.push(u);
        }
        return prices;
    }
};
