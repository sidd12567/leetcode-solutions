class Solution {
    public String restoreString(String s, int[] indices) {
        int si=indices.length;
        char[] ans = new char[si];
        int j=0;
        for(int i:indices){
            ans[i]=s.charAt(j);
            j++;
        }
        String ai="";
        for(int k=0;k<si;k++){
            ai+=ans[k];
        }
        return ai;
    }
}
