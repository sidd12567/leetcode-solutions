class Solution {
    public int singleNumber(int[] nums) {
        HashMap<Integer, Boolean> mp = new HashMap<>();
        int ans=0;
        for(int i:nums){
            if(mp.containsKey(i)){
                mp.put(i, false);
            }
            else{
                mp.put(i, true);
            }
        }
        for(int i:nums){
            if(mp.get(i)==true){
                ans=i;
                break;
            }
        }
        return ans;
    }
}
