class Solution {
    public int[] searchRange(int[] nums, int target) {
        int mn=Integer.MAX_VALUE;
        int mx=Integer.MIN_VALUE;
        for(int i=0;i<nums.length;i++){
            if(nums[i]==target){
                mn=Math.min(i,mn);
                mx=Math.max(i,mx);
            }
        }
        int[] ans = new int[2];
        ans[0]=mn==Integer.MAX_VALUE?-1:mn;
        ans[1]=mx==Integer.MIN_VALUE?-1:mx;
        return ans;
    }
}
