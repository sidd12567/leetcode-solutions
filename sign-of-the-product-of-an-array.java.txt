class Solution {
    public int arraySign(int[] nums) {
        int p=0,n=0;
        for(int i:nums){
            if(i==0){
                return 0;
            }
            else if(i>0){
                p++;
            }
            else{
                n++;
            }
        }
        if(n%2!=0){
            return -1;
        }
       
        return 1;
    }
}
