class Solution {
    public int maximumWealth(int[][] accounts) {
        int ans=0,mx=0;
        for(int i=0;i<accounts.length;i++){
            for(int j=0;j<accounts[i].length;j++){
                ans+=accounts[i][j];

            }
            mx=Math.max(ans,mx);
            ans=0;
        }
        return mx;
    }
}
