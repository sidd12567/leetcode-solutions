class Solution {
public:
    vector<int> arrayRankTransform(vector<int>& arr) {
        map<int,int> mp;
        vector<int> ans(arr.size());
        int j=0;
        vector<int> tmp=arr;
        sort(tmp.begin(),tmp.end());
        for(int i:tmp){
            mp.emplace(i,1+mp.size());
        }
        for(int i=0;i<arr.size();i++){
            ans[i]=mp[arr[i]];
        }
        return ans;
    }
};
